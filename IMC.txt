usuario = {}

def Pedir_Data ():
    nombre = input ('Por favor, ingrese su nombre y apellido: ')
    mail = input ('Por favor, ingrese su mail para enviar los resultados: ')
    peso = input ('Por favor, ingrese su peso en kilogramos: ')
    altura = input ('Por favor, ingrese su altura en metros: ')
    peso = float (peso)
    altura = float (altura)
    usuario['nombre'] = nombre
    usuario['mail'] = mail
    usuario['peso'] = peso
    usuario['altura'] = altura
    
def calculo_IMC ():
    IMC = usuario ['peso']/(usuario ['altura']**2)
    print('Su indice de masa corporal es: '+IMC)
    if IMC < 18.5: 
        print ('Su peso es inferior al normal')
    if 18.5 <= IMC <= 24.9: 
        print ('Su peso es normal')  
    if 25 <= IMC <= 29.9: 
        print ('Su peso es superior al normal')
    if IMC > 30: 
        print ('Usted tiene obesidad')
    
Pedir_Data ()
calculo_IMC ()
